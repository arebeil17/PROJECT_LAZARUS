# Load Data
.main	addi	$t0,	$0,		1
		addi	$t1,	$0,		2
		addi	$t2,	$0,		3
		addi	$t3,	$0,		4
		addi	$t4,	$0,		255
		addi	$t5,	$0,		65535
		addi	$t6,	$0,		-1
		addi	$t7,	$0,		-2
		addi	$t8,	$0,		-3
		addi	$t9,	$0,		-4
		
		#
		# Math Tests
		#
		
		# Addition
		add		$s0,	$t2,	$t3		# 2 + 4 = 6
		add		$s1,	$t3,	$t6 	# 4 + -1 = 3
		add		$s2,	$t7,	$t2		# -2 + 3 = 1
		add		$s3,	$t8,	$t7		# -3 + -2 = -5
		addi	$s4,	$t0,	1		# 1 + 1 = 2
		addi	$s5,	$t1,	-1		# 2 + -1 = 1
		addi	$s6,	$t6,	2		# -1 + 2 = 1
		addi	$s7,	$t7,	-2		# -2 + -2 = -4
		addu	$s0,	$t2,	$t3
		addu	$s1,	$t3,	$t6
		addu	$s2,	$t7,	$t2
		addu	$s3,	$t8,	$t7
		addiu	$s4,	$t0,	1
		addiu	$s5,	$t1,	-1
		addiu	$s6,	$t6,	2
		addiu	$s7,	$t7,	-2
		
		# Subtraction
		sub		$s0,	$t0,	$t1		# 1 - 2 = -1
		sub		$s1,	$t1,	$t0		# 2 - 1 = 1
		sub		$s2,	$t2,	$t7,	# 3 - (-2) = 5
		sub		$s3,	$t6,	$t8		# -1 - (-3) = 2
		
		# Multiplication
		mul		$s0,	$t1,	$t2		# 2 * 3 = 6
		mul		$s1,	$t2,	$t7		# 3 * -2 = -6
		mul		$s2,	$t8,	$t2		# -3 * 3 = -9
		mul		$s3,	$t6,	$t7		# -1 * -2 = 2
		mult	$t1,	$t2				# 2 * 3 = 6
		mult	$t2,	$t7				# 3 * -2 = -6
		mult	$t8,	$t2				# -3 * 3 = -9
		mult	$t6,	$t7				# -1 * -2 = 2
		multu	$t1,	$t2				# 2 * 3 = 6
		multu	$t2,	$t7				# 3 * -2 = -4294967302
		multu	$t8,	$t2				# -3 * 3 = -8589934601
		multu	$t6,	$t7				# -1 * -2 = -12884901886
		madd	$t1,	$t2				# 2 * 3 = 6
		madd	$t2,	$t7				# (6) + 3 * -2 = 0
		madd	$t8,	$t2				# (0) + -3 * 3 = -9
		madd	$t6,	$t7				# (-9) + -1 * -2 = -7
		msub	$t1,	$t2				#  (-7) - 2 * 3 = -13
		msub	$t2,	$t7				#  (-13) - 3 * -2 = -7
		msub	$t8,	$t2				#  (-7) - -3 * 3 = 2
		msub	$t6,	$t7				#  (2) - -1 * -2 = 0
		
		#
		# Logic Tests
		#
		
		# AND Tests
		and		$s0,	$t0,	$t1		# 1 & 2 = 0
		and		$s1,	$t2,	$t1		# 3 & 1 = 1
		andi	$s2,	$t3,	4		# 4 & 4 = 4
		andi	$s3,	$t1,	8		# 2 & 8 = 0
		
		# OR Tests
		or		$s0,	$t1,	$t2		# 1 | 2 = 3
		or		$s1,	$t3,	$t3		# 4 | 4 = 4
		ori		$s2,	$t2,	1		# 3 | 1 = 3
		ori		$s3,	$t3,	3		# 4 | 3 = 7
		
		# NOR Tests
		nor		$s0,	$t0,	$t1		# ~(1 | 2) = 0
		nor		$s1,	$0,		$0		# ~(0 | 0) = 1
		
		# XOR Tests
		xor		$s0,	$t1,	$t2		# 2 ^ 3 = 1
		xor		$s1,	$t2,	$t3		# 3 ^ 4 = 7
		xori	$s2,	$t3,	1		# 4 ^ 1 = 5
		xori	$s3,	$t4,	256		# 255 ^ 256 = 511
		
		# Shift Left Tests
		sll		$s0,	$t4,	1		# 255 << 1 = 510
		sll		$s1,	$0,		2		# 0 << 2 = 0
		sllv	$s2,	$t4,	$t0		# 255 << 1 = 510
		sllv	$s3,	$0,		$t1		# 0 << 2 = 0
		
		# Shift Right Tests
		srl		$s0,	$t4,	1		# 255 >> 1 = 127
		srl		$s1,	$0,		2		# 0 >> 2 = 0
		srlv	$s2,	$t4,	$t0		# 255 >> 1 = 127
		srlv	$s3,	$0,		$t1		# 0 >> 2 = 0
			
		# Sign Extend Tests
		seh		$s0,	$t5				# -1
		seh		$s1,	$t4				# 255
		seb		$s2,	$t3				# 4
		seb		$s3,	$t4				# -1
		
		# Set Less Than Tests
		slt		$s0,	$t0,	$t1		# (1 < 2) = 1
		slt		$s1,	$t3,	$t2		# (4 < 3) = 0
		slt		$s2,	$t8,	$t7		# (-3 < -2) = 1
		slt		$s3,	$0,		$t6		# (0 < -1) = 0
		sltu	$s4,	$t0,	$t1		# (1 < 2) = 1
		sltu	$s5,	$t3,	$t2		# (4 < 3) = 0
		sltu	$s6,	$t8,	$t7		# (-3 < -2) = 0
		sltu	$s7,	$0,		$t6		# (0 < -1) = 1
		slti	$s0,	$t0,	2,		# (1 < 2) = 1
		slti	$s1,	$t3,	3,		# (4 < 3) = 0
		slti	$s2,	$t8,	-2,		# (-3 < -2) = 1
		slti	$s3,	$0,		-1,		# (0 < -1) = 0
		sltiu	$s4,	$t0,	2,		# (1 < 2) = 1
		sltiu	$s5,	$t3,	3,		# (4 < 3) = 0
		sltiu	$s6,	$t8,	-2,		# (-3 < -2) = 0
		sltiu	$s7,	$0,		-1,		# (0 < -1) = 1
				
		# Move Tests
		movn	$s0,	$t0,	$0		# No Move
		movn	$s1,	$t1,	$t0		# Move
		movz	$s2,	$t2,	$0		# Move
		movz	$s3,	$t3,	$t0		# No Move
				
		# Rotate Right Tests
		rotr	$s0,	$t1,	1		# 1
		rotr	$s1,	$t2,	2		# -1073741824
		rotrv	$s2,	$t1,	$t0		# 1
		rotrv	$s3,	$t2,	$t1		# -1073741824
		
		# Arithmatic Shifts
		sra		$s0,	$t0,	1		# 0
		sra		$s1,	$t6,	2		# -1073741825
		srav	$s2,	$t0,	$t0		# 0
		srav	$s3,	$t6,	$t1		# -1073741825
